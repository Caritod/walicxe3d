26 !> @brief First order Lax-Friefrichs scheme
27 subroutine laxfriedrichs ()
28 
29  use parameters
30  use globals
31  use tictoc
32  implicit none
33 
34  integer :: nb, bid, lev, i, j, k, mark
35  real :: pvars(neqtot)
36 
37  ! Exchange boundaries to fill ghost cells (all blocks)
38  call boundary(1, u)
39 
40  ! Update primitives in ghost cells
41  call calcprimsall(u, prim, cells_ghost)
42 
43  ! Integrate all local blocks
44  do nb=1,nbmaxproc
45  bid = localblocks(nb)
46  if (bid.ne.-1) then
47 
48  call tic(mark)
49 
50  ! Compute physical fluxes for all physical cells, plus one additional
51  ! ghost cell to each side
52  do i=0,ncells_x+1
53  do j=0,ncells_y+1
54  do k=0,ncells_z+1
55  pvars(:) = prim(nb,:,i,j,k)
56  call prim2fluxes(pvars, dim_x, f(:,i,j,k))
57  call prim2fluxes(pvars, dim_y, g(:,i,j,k))
58  call prim2fluxes(pvars, dim_z, h(:,i,j,k))
59  end do
60  end do
61  end do
62 
63  ! Compute UPs using Lax-Friedrich's upwind scheme
64  call meshlevel(bid,lev)
65  do i=1,ncells_x
66  do j=1,ncells_y
67  do k=1,ncells_z
68  up(nb,:,i,j,k) = &
69  ( u(nb,:,i+1,j,k) + u(nb,:,i-1,j,k) &
70  + u(nb,:,i,j+1,k) + u(nb,:,i,j-1,k) &
71  + u(nb,:,i,j,k+1) + u(nb,:,i,j,k-1) ) / 6.0 &
72  - 0.5*dt/dx(lev)*(f(:,i+1,j,k)-f(:,i-1,j,k)) &
73  - 0.5*dt/dy(lev)*(g(:,i,j+1,k)-g(:,i,j-1,k)) &
74  - 0.5*dt/dz(lev)*(h(:,i,j,k+1)-h(:,i,j,k-1))
75  end do
76  end do
77  end do
78 
79 ! write(logu,*) "Integrated block", bID, "in", nicetoc(mark)
80 
81  end if
82  end do
83 
84 end subroutine laxfriedrichs
