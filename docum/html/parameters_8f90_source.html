67  ! Mesh Geometry

69  !> Method used to specify mesh geometry
70  ! There are currently two ways to specify the mesh geometry:
71  ! 1) MESH_AUTO: the user specifies max-level resolution.
72  ! This will automatically calculate the number of root blocks
73  ! and number of refinement levels
74  ! 2) MESH_MANUAL: the user specifies the number and geometry of the
75  ! root blocks and the number of refinement levels.
76  ! The max-level resolution will then be given by:
77  ! dx(max) = phys_size / (ncells_x*2^(maxlev-1))
78  ! In both cases, the requested grid geometry must match the aspect
79  ! ratio of the physical box dimensions.
80  integer, parameter :: mesh_method = MESH_AUTO
81 
82  ! Only one of the following group of parameters will be used:
83 
84  ! 1) MESH_AUTO
85  ! Specify the maximum number of cells desired at the highest refinement
86  ! level
87  ! > MUST BE POWERS OF TWO! <
88  integer, parameter :: p_maxcells_x = 128
89  integer, parameter :: p_maxcells_y = 128
90  integer, parameter :: p_maxcells_z = 128
91 
92  ! -- OR --
93 
94  ! 2) MESH_MANUAL
95  ! Specify number of root blocks along each dimension and number of
96  ! refinement levels
97  integer, parameter :: p_nbrootx = 1
98  integer, parameter :: p_nbrooty = 1
99  integer, parameter :: p_nbrootz = 1
100  integer, parameter :: p_maxlev = 5

102  ! Other Mesh Parameters
103 
104  !> Number of cells per block (per dimension)
105  ! This specifies the number of cells per block along each dimension
106  ! (i.e., along each side of the cube that is a block)
107  ! This is a tradeoff between 1) having a few large blocks, which reduces
108  ! the amount of inter-process communication but also makes the mesh
109  ! refinement inefficient (forcing the code to refine large volumes),
110  ! and 2) having a lot of small blocks, which makes the AMR very efficient
111  ! but increases the amount of communication.
112  ! MUST be a power of two. A good value is 16.
113  integer, parameter :: ncells_block = 16
114 
115  ! Refinement / Coarsening thresholds
116  ! These are the maximum relative gradients required to mark a block for
117  ! either refinement or coarsening. Note: refineThres must be larger than
118  ! twice coarseThres. Good values are 0.3 and 0.03.
119  real, parameter :: refineThres = 0.3 !< Refinement threshold
120  real, parameter :: coarseThres = 0.03 !< Coarsening threshold
