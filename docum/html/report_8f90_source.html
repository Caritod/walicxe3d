26 !> @brief Prints out the "main" progress report (after each iteration)

33 
34  character(30) :: timestr, dtstr

36  call nicetime(time*t_sc, timestr)

44  write(logu,'(1x,a,a)') "time = ", trim(timestr)

47  write(logu,*) "============================================"

49  trim(timestr), "; dt=", trim(dtstr), "; blocks=", nblocal, "/", nbactive
50  write(logu,*) "============================================"
51  write(logu,*) ""
