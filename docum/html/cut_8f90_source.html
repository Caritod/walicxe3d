215 !> @brief Returns the local cell plane of this block that intersects
216 !! the given plane, or -1 if it doesn't cut it
217 subroutine getcellplane (bID, cut_axis, cut_loc, plane)
218 
219  use parameters
220  use globals
221  implicit none
222 
223  integer, intent(in) :: bid
224  integer, intent(in) :: cut_axis
225  real, intent(in) :: cut_loc
226  integer, intent(out) :: plane
227 
228  real :: xl, xh, yl, yh, zl, zh, bl, bh
229  integer :: ilev
230 
231  call bounds(bid, xl, xh, yl, yh, zl, zh)
232  call meshlevel(bid, ilev)
233 
234  if (cut_axis.eq.axis_x) then
235  bl = xl
236  bh = xh
237  else if (cut_axis.eq.axis_y) then
238  bl = yl
239  bh = yh
240  else if (cut_axis.eq.axis_z) then
241  bl = zl
242  bh = zh
243  else
244  write(*,'(1x,a)') "Invalid cut axis!!"
245  end if
246 
247  ! If block intersects cut plane, determine intersection cell plane.
248  ! Otherwise, return -1.
249  if ((cut_loc.ge.bl).and.(cut_loc.lt.bh)) then
250  plane = int((cut_loc-bl)/(dx(ilev))) + 1
251  else
252  plane = -1
253  end if
254 
255  return
256
