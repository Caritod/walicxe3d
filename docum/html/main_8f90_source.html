26 !> @brief Walicxe3D Main Program
27 
28 !> This is the main program unit of the Walicxe3D code. It manages all the
29 !! high-level operations of the code, which include initializations, initial
30 !! flow conditions, adaptive mesh refinements, flow variables integration,
31 !! boundary conditions calculation, parallel load balancing.
32 program walicxe3d
33 
34  use parameters ! Code parameters
35  use globals ! Runtime global variables
36  use tictoc ! Time-measuring library
37  implicit none ! ALWAYS mandatory
38 
39  ! Timing mark
40  call tic(start_mark)
41 
42  ! Initialize global arrays and variables
43  call initmain()
44 
45  ! Initialize base grid, and refine it (cold start)
46  call basegrid()
47 
48  ! Impose initial conditions (cold start) or load them from file (warm start)
49  if (dowarm) then
50  call warmstart()
51  else
52  call initflow()
53  end if
54 
55  ! Distribute base load among all processors
56  call loadbalance()
57 
58  ! Update primitives
59  call updateprims()
60 
61  ! Write initial condition to disk
62  if (nextout.eq.0) then
63  call output(0)
64  nextout = 1
65  end if
66 
67  ! Main Loop
68 ! do while(.false.)
69  do while (time.le.tfin/t_sc)
70 
71  it = it + 1
72 
73  write(logu,'(a)') "================================================================================"
74  write(logu,'(1x,a,i0)') "Starting Iteration " , it
75  write(logu,'(1x,a)') stamp()
76  write(logu,'(a)') "================================================================================"
77 
78  ! Hydro Solver
79  call hydrosolver()
80 
81  ! Update primitives in all blocks
82  call updateprims()
83 
84  ! Radiative cooling
85  call cooling()
86 
87  ! Update AMR grid
88  call admesh()
89 
90  ! Load balance
91  call loadbalance()
92 
93  ! Data output (if scheduled)
94  if (time.ge.nextout*dtout/t_sc) then
95  call output(nextout)
96  nextout = nextout + 1
97  end if
98 
99  ! Report progress
100  call main_report()
101 
102  ! Everyone stop here
103  call mpi_barrier(mpi_comm_world, ierr)
104 
105  end do
106 
107  ! Deallocate globals and terminate execution
108  write(logu,*) ""
109  write(logu,'(a)') "================================================================================"
110  write(logu,'(a)') stamp()
111  write(logu,'(a)') 'Execution complete!'
112  write(logu,'(a,a)') 'Total elapsed time: ', nicetoc(start_mark)
113  call deinit()
114 
115 end program walicxe3d
