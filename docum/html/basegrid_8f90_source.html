26 !> @brief Initializes root blocks and refines the mesh to prepare it for ICs
27 !> @details This is a main-level routine. Calculates the mesh geometry
28 !! (how many root blocks) and number of levels to meet the required resolution,
29 !! and refines the IC zone.
30 subroutine basegrid ()
31 
32  use parameters
33  use globals
34  use tictoc
35  implicit none
36 
37  ! Local variables
38  integer :: maxlevx, maxlevy, maxlevz, smalldim
39  integer :: nb, bid, ilev, x, y, z, mark
40  real :: smallsize
41 
42  call tic(mark)
43 
44  write(logu,*) ""
45  write(logu,*) "============================================"
46  write(logu,'(1x,a)') " Initializing the base grid ..."
47  write(logu,*) "============================================"
48  write(logu,*) ""
49 
50  ! Every rank calculates the base grid geometry, based on the selected method
51  if (mesh_method.eq.mesh_auto) then
52 
53  maxcells_x = p_maxcells_x
54  maxcells_y = p_maxcells_y
55  maxcells_z = p_maxcells_z
56 
57  ! Number of root blocks
58  smalldim = min(maxcells_x,maxcells_y,maxcells_z)
59  nbrootx = maxcells_x / smalldim
60  nbrooty = maxcells_y / smalldim
61  nbrootz = maxcells_z / smalldim
62 
63  ! Calculate number of levels
64  maxlevx = int(log(float(maxcells_x)/nbrootx/ncells_x)/log(2.)+1)
65  maxlevy = int(log(float(maxcells_y)/nbrooty/ncells_y)/log(2.)+1)
66  maxlevz = int(log(float(maxcells_z)/nbrootz/ncells_z)/log(2.)+1)
67  maxlev = max(maxlevx,maxlevy,maxlevz)
68 
69 
70  else if (mesh_method.eq.mesh_manual) then
71 
72  maxlev = p_maxlev
73  nbrootx = p_nbrootx
74  nbrooty = p_nbrooty
75  nbrootz = p_nbrootz
76 
77  ! Number of max-resolution cells
78  maxcells_x = ncells_x * 2**(maxlev-1)
79  maxcells_y = ncells_y * 2**(maxlev-1)
80  maxcells_z = ncells_z * 2**(maxlev-1)
81 
82  end if
83 
84  ! The maximum allowable number of levels is 10 at the moment, due to the
85  ! fact that bIDs are handled as 4-byte (signed) integers. With 10 levels,
86  ! one can have up to 14 root blocks.
87  if (maxlev.gt.10) then
88  write(logu,*) ""
89  write(logu,'(1x,a)') "Desired finest-level resolution would require more than 10 levels!"
90  write(logu,'(1x,a)') "***ABORTING***"
91  call clean_abort(error_too_many_levs)
92  end if
93 
94  ! Check if physical sizes conform to desired max-resolution cell numbers
95  smallsize = min(xphystot, yphystot, zphystot)
96  if ((maxcells_x/smalldim.ne.xphystot/smallsize).or.&
97  (maxcells_y/smalldim.ne.yphystot/smallsize).or.&
98  (maxcells_z/smalldim.ne.zphystot/smallsize)) then
99  write(logu,*)
100  write(logu,'(a)') "ERROR: The physical size of the simulation box has"//&
101  " a different aspect ratio than the requested number of cells!"
102  write(logu,'(a,f4.1,a,f4.1,a,f4.1)') "Physical size aspect ratio: ",&
103  xphystot/smallsize, " : ", yphystot/smallsize, " : ", zphystot/smallsize
104  write(logu,'(a,i3,a,i3,a,i3)') "Cell number aspect ratio: ",&
105  maxcells_x/smalldim, " : ", maxcells_y/smalldim, " : ", maxcells_z/smalldim
106  write(logu,'(a)') "> Modify parameters.f90"
107  call clean_abort(error_basegrid_bad_aspect)
108  else
109  write(logu,'(1x,a,i2,a,i2,a,i2)') "Grid geometry (root blocks): ", &
110  nbrootx, " x ", nbrooty, " x ", nbrootz
111  write(logu,'(1x,a,i3)') "Number of root blocks: ", nbrootx*nbrooty*nbrootz
112  write(logu,'(1x,a,i2)') "Number of refinement levels: ", maxlev
113  end if
114 
115  ! Allocate and initialize grid spacing at each level (code units)
116  allocate( dx(maxlev) )
117  allocate( dy(maxlev) )
118  allocate( dz(maxlev) )
119  do ilev=1,maxlev
120  dx(ilev) = xphystot / (ncells_x * nbrootx * 2.0**(ilev-1)) / l_sc
121  dy(ilev) = yphystot / (ncells_y * nbrooty * 2.0**(ilev-1)) / l_sc
122  dz(ilev) = zphystot / (ncells_z * nbrootz * 2.0**(ilev-1)) / l_sc
123  end do
124 
125  ! Calculate total number of blocks per level
126  allocate( nblockslev(maxlev) )
127  do ilev=1,maxlev
128  nblockslev(ilev) = nbrootx*nbrooty*nbrootz*8**(ilev-1)
129  end do
130 
131  ! Calculate number of blocks along each dimension at each level
132  allocate( nbx(maxlev) )
133  allocate( nby(maxlev) )
134  allocate( nbz(maxlev) )
135  do ilev=1,maxlev
136  nbx(ilev) = nbrootx*2**(ilev-1)
137  nby(ilev) = nbrooty*2**(ilev-1)
138  nbz(ilev) = nbrootz*2**(ilev-1)
139  end do
140 
141  ! Calculate the bIDs of the first and last blocks at each level
142  allocate( minid(maxlev) )
143  allocate( maxid(maxlev) )
144  minid(1) = 1
145  maxid(1) = nblockslev(1)
146  do ilev=2,maxlev
147  minid(ilev) = maxid(ilev-1) + 1
148  maxid(ilev) = maxid(ilev-1) + nblockslev(ilev)
149  end do
150 
151  nblocal = 0
152  nbactive = nbrootx*nbrooty*nbrootz
153 
154  ! ==========================
155  ! The following is only performed in cold starts
156 
157  if (.not.dowarm) then
158 
159  ! Activate root blocks and initialize block registry
160  write(logu,'(1x,a)') "> Creating root blocks ..."
161 
162  ! Calculate bID of root blocks and register them in master's block list
163  if (rank.eq.master) then
164  nb = 1
165  do z = 1,nbrootz
166  do y = 1,nbrooty
167  do x = 1,nbrootx
168 
169  ! bID is sensitive to numbering order
170  bid = 1+(x-1)+(y-1)*nbrootx+(z-1)*nbrootx*nbrooty
171 
172  ! Register block
173  localblocks(nb) = bid
174 
175  nb = nb + 1
176  nblocal = nblocal + 1
177 
178  end do
179  end do
180  end do
181  end if
182 
183  ! Synchronize block lists
184  call syncblocklists()
185 
186  else
187 
188  write(logu,*) ""
189  write(logu,'(1x,a)') "> Skipping root block creation (warm start)"
190 
191  end if
192 
193  ! ==========================
194 
195  write(logu,*) ""
196  write(logu,'(1x,a,a)') "> Created base grid in ", nicetoc(mark)
197  write(logu,*) ""
198 
199 end subroutine basegrid
